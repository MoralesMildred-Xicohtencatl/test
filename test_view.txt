SELECT DISTINCT SOL.num_solicitud_tramite ,COM.folio_comprobante  ,ifnull(PAG.mto_importe_siniestral,0) mto_importe_siniestral ,SOL.num_reclamacion ,COM.des_estatus_comprobante ,VAL.mto_reclamado ,SOL.des_motivo_terminacion ,COM.ts_alta_hdfs ts_alta_dlk   from  z_gmm.alm_comprobante_ecs COM  LEFT JOIN  z_gmm.alm_solicitud_ecs SOL    ON COM.FOLIO_SOLICITUD = SOL.FOLIO_SOLICITUD   AND COM.ID_SOLICITUD = SOL.ID_SOLICITUD LEFT JOIN z_gmm.alm_solicitud_pago_ecs PAG    ON COM.ID_SOLICITUD = PAG.ID_SOLICITUD   AND COM.NUM_ORDEN_PAGO = PAG.NUM_ORDEN_PAGO LEFT JOIN (   select COM.folio_comprobante,COM.folio_solicitud,COM.id_solicitud,sum(COM.mto_reclamado_mon_nal) mto_reclamado   from z_gmm.alm_comprobante_ecs COM   where cve_sistema_int = 'INFO'   group by COM.folio_comprobante,COM.folio_solicitud,COM.id_solicitud ) VAL   on COM.folio_comprobante= VAL.folio_comprobante     and COM.folio_solicitud = VAL.folio_solicitud     and COM.id_solicitud = VAL.id_solicitud      where COM.cve_sistema_int = 'INFO' 	and COM.id_solicitud = '0105109458'
